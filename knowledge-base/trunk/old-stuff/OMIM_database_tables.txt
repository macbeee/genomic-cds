disorder VARCHAR(150) NOT NULL,
gene_symbol VARCHAR(100),
gene_MIMno VARCHAR(100),
cytogenetic_loc VARCHAR(100)
);

LOAD DATA LOCAL INFILE '/****path***'
INTO TABLE morbidmap
FIELDS TERMINATED BY '|'
